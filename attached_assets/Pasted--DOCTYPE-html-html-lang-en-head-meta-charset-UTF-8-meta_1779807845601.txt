<script>

/* CHATS */
let chats = [
  {id:"coach", name:"Coach Muhavi", unread:2},
  {id:"james", name:"James Otieno", unread:1},
  {id:"scout", name:"Scout Office", unread:0}
];

/* MESSAGES */
let messages = {
  coach:[{text:"Training at 6AM", me:false}],
  james:[{text:"Welcome player", me:false}],
  scout:[{text:"Scout report ready", me:false}]
};

let active = null;

/* NOTIFICATION */
function updateNotif(){
  let total = chats.reduce((sum,c)=>sum + c.unread,0);
  document.getElementById("notifbar").innerText =
    `🔔 ${total} unread messages`;
}

/* RENDER LIST */
function renderList(){

  const box = document.getElementById("list");
  const search = document.getElementById("search").value.toLowerCase();

  box.innerHTML = "";

  chats
  .filter(c=>c.name.toLowerCase().includes(search))
  .forEach(c=>{

    const div = document.createElement("div");
    div.className = "item" + (active===c.id ? " active":"");

    div.innerHTML = `
      <div>
        <div>${c.name}</div>
        <div style="font-size:11px;color:#777">
          Chat
        </div>
      </div>

      <div style="display:flex;gap:6px;align-items:center;">
        ${c.unread > 0 ? `<span class="badge">${c.unread}</span>` : ""}
        <button class="del" onclick="deleteChat('${c.id}')">🗑</button>
      </div>
    `;

    div.onclick = ()=>openChat(c.id);

    box.appendChild(div);
  });

  updateNotif();
}

/* OPEN CHAT */
function openChat(id){
  active = id;

  // mark read
  let chat = chats.find(c=>c.id===id);
  if(chat){
    chat.unread = 0;
  }

  document.getElementById("header").innerText = id.toUpperCase();

  renderMessages();
  renderList();
}

/* DELETE CHAT */
function deleteChat(id){
  chats = chats.filter(c=>c.id!==id);
  delete messages[id];

  if(active===id){
    active = null;
    document.getElementById("messages").innerHTML = "";
    document.getElementById("header").innerText = "Select chat";
  }

  renderList();
}

/* MESSAGES */
function renderMessages(){

  const box = document.getElementById("messages");
  box.innerHTML = "";

  if(!active) return;

  (messages[active]||[]).forEach(m=>{
    const div = document.createElement("div");
    div.className = "msg " + (m.me ? "me":"");
    div.textContent = m.text;
    box.appendChild(div);
  });

  box.scrollTop = box.scrollHeight;
}

/* SEND */
function send(){

  const input = document.getElementById("input");
  const text = input.value.trim();

  if(!text || !active) return;

  messages[active].push({text,me:true});

  input.value = "";

  renderMessages();
}

/* INIT */
renderList();
updateNotif();

</script>
