```typescript
// app/api/support/inbound-email/route.ts
// Configure SendGrid Inbound Parse (or equivalent) to POST here

export async function POST(req: Request) {
  const formData = await req.formData()
  const fromEmail = formData.get('from')
  const subject = formData.get('subject')
  const body = formData.get('text')

  verifyInboundWebhookSignature(req) // reject if signature invalid

  const existingUser = await findUserByEmail(fromEmail)
  const existingTicket = await findOpenTicketBySender(fromEmail)

  if (existingTicket) {
    await appendMessage(existingTicket.id, { senderType: 'user', body, sentVia: 'email' })
    return Response.json({ success: true })
  }

  const priority = detectPriority(subject + body)
  const ticketRef = await createDoc('support_tickets', {
    senderUserId: existingUser?.uid ?? null,
    senderEmail: fromEmail,
    senderName: extractNameFromSignature(body) ?? fromEmail,
    source: 'email',
    subject,
    status: 'open',
    priority,
    slaDeadline: addHours(new Date(), priority === 'high' ? 1 : 4),
    accountProvisioned: false,
    createdAt: serverTimestamp(),
  })

  await appendMessage(ticketRef.id, { senderType: 'user', body, sentVia: 'email' })
  return Response.json({ success: true })
}

function detectPriority(text: string): 'low' | 'medium' | 'high' {
  const highKeywords = ['payment', 'paid', "can't login", 'charged', 'urgent']
  return highKeywords.some(k => text.toLowerCase().includes(k)) ? 'high' : 'medium'
}
```

```typescript
// app/api/admin/support/provision-account/route.ts
export async function POST(req: Request) {
  const agent = await verifyAgentToken(req) // admin | support_agent only
  const { ticketId, email, name, role } = await req.json()

  const ticket = await getTicket(ticketId)
  if (ticket.senderEmail !== email) throw new Error('Email mismatch with ticket')

  const existing = await findUserByEmail(email)
  if (existing) throw new Error('Account already exists for this email')

  // Identity Toolkit REST API — consistent with rest of platform, no Admin SDK
  await fetch(`https://identitytoolkit.googleapis.com/v1/accounts:sendOobCode?key=${FIREBASE_API_KEY}`, {
    method: 'POST',
    body: JSON.stringify({
      requestType: 'EMAIL_SIGN_IN',
      email,
      continueUrl: `${BASE_URL}/auth/complete?ticketId=${ticketId}`,
      canHandleCodeInApp: true,
    }),
  })

  const reqRef = await addDoc('provisioning_requests', {
    ticketId, email, name, assignedRole: role,
    triggeredBy: agent.uid, status: 'link_sent',
    expiresAt: addHours(new Date(), 24), createdAt: serverTimestamp(),
  })

  await updateDoc(`support_tickets/${ticketId}`, { accountProvisioned: false })
  return Response.json({ success: true, requestId: reqRef.id })
}
```

```typescript
// app/api/auth/complete-provisioning/route.ts
export async function POST(req: Request) {
  const idToken = req.headers.get('Authorization')?.split(' ')[1]
  const decoded = await verifyIdToken(idToken)
  const { ticketId } = await req.json()

  const provReq = await getDocByField('provisioning_requests', 'ticketId', ticketId)
  if (provReq.email !== decoded.email) throw new Error('Email mismatch — security check failed')
  if (provReq.status !== 'link_sent') throw new Error('Link already used, expired, or revoked')

  await batchWrite([
    { ref: `users/${decoded.uid}`, data: {
      uid: decoded.uid, email: decoded.email, displayName: provReq.name,
      role: provReq.assignedRole, accountStatus: 'provisional',
      provisionedFromTicketId: ticketId, onboardingCompleted: [],
      createdAt: serverTimestamp(),
    }},
    { ref: `${roleCollection(provReq.assignedRole)}/${decoded.uid}`, data: minimalStarterDoc(provReq) },
    { ref: `support_tickets/${ticketId}`, data: { accountProvisioned: true, provisionedUserId: decoded.uid } },
    { ref: `provisioning_requests/${provReq.id}`, data: { status: 'completed' } },
  ])

  return Response.json({ role: provReq.assignedRole, redirectTo: roleToDashboard(provReq.assignedRole) })
}
```

```typescript
// app/api/admin/marketing/segments/route.ts
async function buildSegmentQuery(criteria: SegmentFilters) {
  let q = query(collection(db, 'users'))
  if (criteria.role) q = query(q, where('role', '==', criteria.role))
  if (criteria.lastActiveDaysAgo) {
    const cutoff = subDays(new Date(), criteria.lastActiveDaysAgo)
    q = query(q, where('lastActiveAt', '<=', cutoff))
  }
  if (criteria.geography?.country) {
    q = query(q, where('country', '==', criteria.geography.country))
  }
  return q
}

// Cloud Function — nightly
```

### Step 2 — Campaign Send (With Suppression + Frequency Cap)

```typescript
// app/api/admin/marketing/campaigns/[id]/send/route.ts
export async function POST(req, { params }) {
  const manager = await verifyAgentToken(req) // admin | marketing_manager
  const campaign = await getCampaign(params.id)
  const segment = await getSegment(campaign.segmentId)
  const recipients = await resolveSegmentMembers(segment)

  const eligible = []
  for (const user of recipients) {
    if (await isSuppressed(user.email, user.phone)) continue
    if (await exceedsFrequencyCap(user.uid)) continue // max 2 sends per 7 days
    eligible.push(user)
  }

  let sentCount = 0
  for (const user of eligible) {
    if (['email', 'both'].includes(campaign.channel)) {
      await sendCampaignEmail(user, campaign, trackingPixel(campaign.id, user.uid))
    }
    if (['sms', 'both'].includes(campaign.channel)) {
      await sendCampaignSms(user, campaign) // via existing Africa's Talking integration
    }
    await logSendEvent(campaign.id, user.uid)
    sentCount++
  }

  await updateDoc(`marketing_campaigns/${params.id}`, {
    status: 'sent', sentAt: serverTimestamp(), 'analytics.sent': sentCount,
  })
}

async function isSuppressed(email: string, phone?: string): Promise<boolean> {
  const emailHash = sha256(normalize(email))
  const doc = await getDoc(`suppression_list/${emailHash}`)
  return doc.exists()
}

async function exceedsFrequencyCap(userId: string): Promise<boolean> {
  const recentSends = await countSendsLast7Days(userId)
  return recentSends >= 2
}
```

## Shared Requirement — Non-Negotiable Before Either Module Ships

```
The suppression_list collection must be checked by BOTH:
  - Marketing campaign sends (Part B, Step 2)
  - Any future support-side "do not contact" action

A user who tells a support agent "stop contacting me" and a user 
who clicks "unsubscribe" on a marketing email must hit the exact 
same suppression mechanism. Build and test this shared check first, 
before either module's send/reply functionality goes live.
```
